import fs from "node:fs";
import path from "node:path";

export const rootDir = path.resolve("..");
export const exclude = ["pruned", "node_modules"];

export const getProjects = (dir) =>
  fs.readdirSync(dir)
    .filter(f => !f.startsWith(".") && fs.statSync(path.join(dir, f)).isDirectory())
    .map(f => {
      try {
        return JSON.parse(fs.readFileSync(path.join(dir, f, "package.json"), "utf-8"));
      } catch { return null; }
    })
    .filter(pkg => pkg?.homepage);

export const activeProjects = fs.readdirSync(rootDir)
  .filter(f => !exclude.includes(f) && !f.startsWith(".") && fs.statSync(path.join(rootDir, f)).isDirectory())
  .map(f => {
    try {
      return JSON.parse(fs.readFileSync(path.join(rootDir, f, "package.json"), "utf-8"));
    } catch { return null; }
  })
  .filter(pkg => pkg?.homepage);

export const prunedProjects = getProjects(path.join(rootDir, "pruned"));

Welcome to my website.

{activeProjects.length > 0 && (
  <>
    <h2>Active</h2>
    <ul>
      {activeProjects.map((p) => (
        <li><a href={p.homepage}>{p.name}</a></li>
      ))}
    </ul>
  </>
)}

<h2>Pruned</h2>

<ul>
  {prunedProjects.map((p) => (
    <li><a href={p.homepage}>{p.name}</a></li>
  ))}
</ul>
